from luxalgo_momentum_oscillator import AdaptiveMomentumOscillator
from luxalgo_momentum_oscillator_advanced import (
    SignalConfirmation, DivergenceValidator, TrendRegimeDetector
)
import pandas as pd

# Load your data (OHLCV format)
df = pd.read_csv('btc_daily.csv', index_col='timestamp', parse_dates=True)

# Create oscillator
osc = AdaptiveMomentumOscillator(
    data_length=20,              # Max delta period
    smoothing_length=10,         # AMA smoothing
    divergence_length=14,        # Extrema window
)

# Analyze everything
df_momentum, signals, divergences = osc.analyze(df)

print(f"✓ Signals: {len(signals)}")
print(f"✓ Divergences: {len(divergences)}")

# Check signal quality
for sig in signals[-3:]:
    quality = SignalConfirmation.calculate_signal_quality(sig, df_momentum, idx)
    print(f"Quality: {quality.overall_quality:.0f}%")

# Validate divergences
validator = DivergenceValidator()
for div in divergences[-2:]:
    result = validator.validate_divergence(div, df_momentum)
    if result['valid']:
        print(f"✓ {div.divergence_type.value} confirmed!")

# Check trend regime
regime = TrendRegimeDetector.classify_regime(df_momentum)
print(f"Regime: {regime['regime'].upper()}")

HOW IT WORKS (4 STEPS)

Step 1: CALCULATE MAXIMUM DELTA
  Find largest price move in last N bars
  Use as normalization scale

Step 2: NORMALIZE MOMENTUM
  Momentum = Price Change / Max Delta
  Result: 0-centered comparable value

Step 3: APPLY ADAPTIVE SMOOTHING
  Kaufman's AMA:
  - Fast in trends (high efficiency ratio)
  - Slow in ranges (low efficiency ratio)
  - Reduces lag while filtering noise

Step 4: GENERATE SIGNALS
  Crossovers: Momentum vs Signal Line
  Zero Crosses: Bullish/Bearish turns
  Divergences: Reversal setups

SIGNAL TYPES EXPLAINED

BULLISH
  Momentum crosses above signal/zero
  Accelerating upward momentum
  Entry signal
  Strength: 0-100%

BEARISH
  Momentum crosses below signal/zero
  Accelerating downward momentum
  Exit/short signal
  Strength: 0-100%

ACCELERATION
  Momentum increasing
  Confirms trend continuation
  Hold position

DECELERATION
  Momentum decreasing
  May precede reversal
  Take profit signal

DIVERGENCE_BULLISH
  Price lower low, Momentum higher low
  Hidden bullish setup
  High-quality reversal

DIVERGENCE_BEARISH
  Price higher high, Momentum lower high
  Hidden bearish setup
  High-quality reversal

TROUBLESHOOTING

Q: Too many false signals?
A: 1. Increase data_length (20→25)
   2. Increase smoothing_length (10→14)
   3. Filter by quality (>70%)
   4. Add regime confirmation

Q: Signals too slow?
A: 1. Decrease data_length (20→15)
   2. Decrease smoothing_length (10→7)
   3. Use signal line instead of zero
   4. Add acceleration signals

Q: Few divergences?
A: 1. Decrease divergence_length (14→10)
   2. Lower severity threshold
   3. More price history needed
   4. Check data quality
